#include<Servo.h>
Servo myservo;
int pm=A0;
int pm_read;
int trigpin=3;
int echopin=4;

void setup()
{
  myservo.attach(2);
  myservo.write(180);
  pinMode(trigpin,OUTPUT);
  pinMode(echopin,INPUT);
  Serial.begin(9600);

  
}
void loop()
{
   digitalWrite(trigpin,LOW);
  delayMicroseconds(2);
  digitalWrite(trigpin,HIGH);
  delayMicroseconds(10);
  digitalWrite(trigpin,LOW);
  long duration=pulseIn(echopin,HIGH);
  int distance=duration/58;
  Serial.println(distance);
  delay(500);

  if(distance<30)
  {
    myservo.write(90);
    

  }



  else
  {
    delay(1000);
    myservo.write(180);
    
  }

  

}
